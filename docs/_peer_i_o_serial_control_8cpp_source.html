176 int PeerIOSerialControl::ValueTo7bits(int From8BitValue) {
177  byte lByte = lowByte(From8BitValue);
178  byte hByte = highByte(From8BitValue);
179  if ( From8BitValue > 0x3FFF ) return -1; // Value is too big for a 14-bit Value
180  hByte = hByte << 1; // Make Room on hByte for bit-7 of lByte
181  bitWrite(hByte,0,bitRead(lByte,7)); // Transfer lByte<7> onto hByte<0>
182  if ( From8BitValue > 0x7F ) {
183  bitSet(hByte,7); // Value > 7-bits so Set 'END_BIT' @ hByte
184  bitClear(lByte,7);
185  } else {
186  bitSet(lByte,7); // Value <= 7-bits so Set 'END_BIT' @ lByte
187  bitClear(hByte,7);
188  }
